May 10th, 2008

Listen 80
Listen 88 #===============================================================================
<VirtualHost 127.0.0.1:88>
DocumentRoot “D:EmpireServerphpapache2.2htdocs”
</VirtualHost>
<Directory “D:EmpireServerphpapache2.2htdocs”>
# # Possible values for the Options directive are “None”, “All”,
